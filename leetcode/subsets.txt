class Solution {
    public List<List<Integer>> subsets(int[] nums) {
        List<List<Integer>> result = new ArrayList<>();
        getSubsets(0,result, new ArrayList<>(),nums);
        return result;
    }

    private void getSubsets(int i , List<List<Integer>> result, List<Integer> ans, int[] nums ){
        if(i >= nums.length){
            result.add(new ArrayList(ans));
            return;
        }

        ans.add(nums[i]);
        getSubsets(i+1, result,ans,nums);

        ans.remove(ans.size()-1);
        getSubsets(i+1, result, ans, nums);
    }
}
