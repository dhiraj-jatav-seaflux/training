class Solution {
    public boolean isHappy(int n) {
        Set<Integer> set = new HashSet<>();

        return checkHappy(n,set);
    }

    private boolean checkHappy(int n, Set<Integer> set){
        if(n == 1 || n == 100) return true;

        if(set.contains(n)) return false;

        set.add(n);

        int digit = 0;

        while(n>0){
            int lastDigit = n%10;
            digit = digit + (lastDigit*lastDigit);
            n = n/10;
        }

        return checkHappy(digit,set);
    }
}
